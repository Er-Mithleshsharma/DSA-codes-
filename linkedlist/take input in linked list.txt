node *takeinput(){
    int data;
    cin>>data;
    node *head=NULL;
    node *tail=NULL;
    while(data!=-1)
    {
        node *newnode = new node (data);
        if (head == NULL)
        {
            head = newnode;
            tail=newnode;
        }
        else
        {
            tail -> next = newnode;
            tail = tail -> next;
        }
        cin>>data;
    }
    return head;
}
